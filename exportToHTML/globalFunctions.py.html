21       diff = [0, 0, 0]
22       for d in range(3):

24       return math.sqrt(diff[0] ** 2 + diff[1] ** 2 + diff[2] ** 2)

51   # returns the direction and magnitude of pitch and yaw angles between 2 positions using spherical coordinate geometry
52   def getThreeDAngle(cordOne: list, cordTwo: list, xyz: str) -> list:
53       diff = [0, 0, 0]
54       for d in range(3):
55           diff[d] = cordOne[d] - cordTwo[d]
56   
57       # change order of angular calcs when calculating angles relative to x, y, or z
58       if xyz == 'x':
59           diffIdx = [2, 1, 0]
60       elif xyz == 'y':
61           diffIdx = [0, 2, 1]
62       elif xyz == 'z':
63           diffIdx = [0, 1, 2]
64   
65       if diff[diffIdx[1]] != 0:  # prevents ZeroDivisionError
66           pitch = math.atan(diff[diffIdx[0]] / diff[diffIdx[1]])  # relative angle from x to z
67       else:
68           pitch = math.pi / 2
69       if (diff[diffIdx[0]] != 0) or (diff[diffIdx[1]] != 0):  # prevents ZeroDivisionError
70           yaw = math.atan(diff[diffIdx[2]] / math.sqrt(diff[diffIdx[1]] ** 2 + diff[diffIdx[0]] ** 2))  # relative angle from y to x/z
71       else:
72           yaw = math.pi / 2
73   
74       return [pitch, yaw, 0]

77   # return only the magnitude of pitch and yaw angles (in radians) between 2 positions using spherical coordinate geometry
78   def getAbsThreeDAngle(cordOne: list, cordTwo: list, xyz: str) -> list:
79       diff = [0, 0, 0]
80       for d in range(3):
81           diff[d] = abs(cordOne[d] - cordTwo[d])
82   
83       # change order of angular calcs when calculating angles relative to x, y, or z
84       # by default, it will be x
85       if (not xyz) or (xyz == 'x'):
86           diffIdx = [2, 1, 0]
87       elif xyz == 'y':
88           diffIdx = [0, 2, 1]
89       elif xyz == 'z':
90           diffIdx = [0, 1, 2]
91   
92       if diff[diffIdx[1]] != 0:  # prevents ZeroDivisionError
93           pitch = math.atan(diff[diffIdx[0]] / diff[diffIdx[1]])  # relative angle from x to z
94       else:
95           pitch = math.pi / 2
96       if (diff[diffIdx[0]] != 0) or (diff[diffIdx[1]] != 0):  # prevents ZeroDivisionError
97           yaw = math.atan(diff[diffIdx[2]] / math.sqrt(diff[diffIdx[1]] ** 2 + diff[diffIdx[0]] ** 2))  # relative angle from y to x/z
98       else:
99           yaw = math.pi / 2
100  
101      return [pitch, yaw, 0]

104  # returns the angle [pitch, yaw, roll] (in radians) between 2 points. this is a special case for the Vizard setEuler function.
105  def getEulerAngle(cordOne: list, cordTwo: list) -> list:
106      diff = [0, 0, 0]
107      for d in range(3):
108          diff[d] = cordOne[d] - cordTwo[d]
109  
110      if diff[2] != 0:  # prevents ZeroDivisionError
111          pitch = math.degrees(math.atan(diff[0] / diff[2]))  # angle from x to z
112      else:
113          pitch = 90
114      if diff[0] != 0:  # prevents ZeroDivisionError
115          yaw = 90 - math.degrees(math.atan(diff[1] / math.sqrt(diff[0] ** 2 + diff[2] ** 2)))  # angle from y to x/z
116      else:
117          yaw = 90
118      if diff[2] <= 0:
119          yaw = -yaw  # reversed when there's negative difference along the z-axis
120      roll = jointMadness
121  
122      return [pitch, yaw, roll]

148  # resolves spheres (points) bouncing on the edges of collision cuboids (collision rects)
149  # the returned value to set the new cords to depends on the colliding angle and colliding plane of the point
150  # check out the maths here: https://drive.google.com/file/d/1ES6T8RilTcE5Pu7Zhdxfo6R6hvvsViAT/view?usp=drive_link
151  def edgeBounce(resultV: float, angle: list, e: float, angleState: int) -> tuple:
152      if angleState == 0:
153          if (abs(math.degrees(angle[0])) % 90) > 45:
154              return resultV * sin(angle[1] * 2) * sin(angle[0]) * e, resultV * cos(angle[1] * 2) * e, resultV * sin(angle[1] * 2) * cos(angle[0]) * e  # here's why I multiply the angle by 2: https://drive.google.com/file/d/1BeUFv6UuvRvqFxwgApfSc328uGC2fKmZ/view?usp=drive_link
155          else:
156              return resultV * sin(angle[1] * 2) * sin(angle[0]) * e, resultV * cos(angle[1] * 2), resultV * sin(angle[1] * 2) * cos(angle[0]) * e
157      else:
158          if (abs(math.degrees(angle[0])) % 90) < 45:
159              return resultV * sin(angle[1] * 2) * sin(angle[0]) * e, resultV * cos(angle[1] * 2) * e, resultV * sin(angle[1] * 2) * cos(angle[0]) * e  # here's why I multiply the angle by 2: https://drive.google.com/file/d/1BeUFv6UuvRvqFxwgApfSc328uGC2fKmZ/view?usp=drive_link
160          else:
161              return resultV * sin(angle[1] * 2) * sin(angle[0]) * e, resultV * cos(angle[1] * 2), resultV * sin(angle[1] * 2) * cos(angle[0]) * e

182  # returns the dot product between 2 cords (operator can vary) (currently unused)
183  def dotP(pOne: list, pTwo: list, op: str) -> list:
184      if op == '+':
185          return [pOne[0] + pTwo[0], pOne[1] + pTwo[1], pOne[2] + pTwo[2]]
186      elif op == '-':
187          return [pOne[0] - pTwo[0], pOne[1] - pTwo[1], pOne[2] - pTwo[2]]
188      elif op == '*':
189          return [pOne[0] * pTwo[0], pOne[1] * pTwo[1], pOne[2] * pTwo[2]]
190      elif op == '/':
191          return [pOne[0] / pTwo[0], pOne[1] / pTwo[1], pOne[2] / pTwo[2]]

195  def dotPLine(a: list, b: list, c: float, ty: int):
196      ab = dotP(b, a, '-')

198      r = dotP(a, dotP(ab, [m, m, m], '*'), '+')  # resultant cord value

207  # gets the angle (in degrees) and displacement of an object around a bounding sphere (give a radius as disp) from said object to another object (currently only used to make text face towards the camera in some GUIs)
208  def camAnglePos(camCords: list, pCords: list, disp: float) -> (list, list):
209      angleDisp = getThreeDAngle(pCords, camCords, 'y')
210      angle = getEulerAngle(camCords, pCords)
211      multiplier = 1
212      if camCords[2] > pCords[2]:
213          angle[0] -= 180
214          angle[1] -= 180
215          multiplier = -1
216      facingAngle = [angle[0], (angle[1] + 90) * multiplier, 0]
217      if disp != 0:  # only calculate displacement around a sphere when its radius is given
218          pos = pCords[0] - (multiplier * disp * cos(angleDisp[1]) * sin(angleDisp[0])), pCords[1] - (disp * sin(angleDisp[1])), pCords[2] - (multiplier * disp * cos(angleDisp[1]) * cos(angleDisp[0]))
219      else:
220          pos = pCords
221      return facingAngle, pos
