19   # used to get if any hand is colliding with the point of a slider/dial
20   def getCollision(collisionList):
21       collision = False
22       for coll in collisionList:
23           collision = collision or coll  # if any value in 'collisionList' is True, 'collision' will also be True regardless of the other values in 'collisionList'
24       return collision

27   # used to detect for single/double clicks to reset the value of a GUI's reference variable
28   def resetReferenceVar(GUIObj: object, cIdx: int) -> None:
29       """ 
30       :param GUIObj: object of the GUI in question. 
31       :param cIdx: current controller's index position. 
32       """
33       # if time since last reset button click <= 0.25s, increase time since last click
34       if GUIObj.timePressed[cIdx] <= 0.25:
35           GUIObj.timePressed[cIdx] += 1 / renderRate
36   
37       if buttonPressed('reset', GUIObj.cObj[cIdx], cIdx):  # if reset button is pressed
38           if not GUIObj.resetHeld[cIdx]:
39               GUIObj.resetHeld[cIdx] = True
40               if GUIObj.timePressed[cIdx] > 0.25:
41                   GUIObj.resetVar()  # soft reset
42               else:  # time since last reset button click <0.25s, so double-click is registered
43                   GUIObj.resetVar('hard')  # hard reset
44               GUIObj.timePressed[cIdx] = 0  # reset time since last reset button click
45       else:
46           GUIObj.resetHeld[cIdx] = False

119      def interpolate(self) -> None:
120          """ 
121          this method is used to update the value of the reference variable when the slider's pointer is moved. 
122   
123          process: 
124              1. get the displacement of the point along the slider. 
125              2. get a value from 0-1 for how far the point is along the slider. 
126              3. multiply the ratio with the range of the reference variable, and displace it by the minimum value of the reference variable. 
127          """
128          relDist = self.pointer.cords[self.xyz] - self.limits[0]
129          ratio = relDist / self.length
130          self.var = self.min + (self.range * ratio)

132      def resetVar(self, *args):
133          """ 
134          :param args: if this has any value, a hard reset of the variable will occur. otherwise, a soft reset of the variable will occur. 
135   
136          this method is used to reset the value of the reference variable by teleporting the slider's pointer and letting the 'interpolate()' method calculate the new value. 
137          the way this works is 'interpolate()' but in reverse. 
138          """
139          if args:
140              self.pointer.cords[self.xyz] = self.limits[0] + ((self.globalOrigVar + abs(self.min) * -getSign(self.min)) / self.range) * self.length
141          else:
142              self.pointer.cords[self.xyz] = self.limits[0] + ((self.origVar + abs(self.min) * -getSign(self.min)) / self.range) * self.length
143          self.pointer.oldCords = copy.deepcopy(self.pointer.cords)  # set the pointer's velocity to 0 after resetting it, so it doesn't shoot off with insane velocity

145      # set the reference variable to a specific value
146      def setVar(self, var):
147          """ 
148          :param var: new value to set the reference variable to. 
149   
150          this method is used to set the value of the reference variable by teleporting the slider's pointer and letting the 'interpolate()' method calculate the new value. 
151          the way this works is 'interpolate()' but in reverse. 
152          if the reference variable is a vector quantity (list), update the value of 'globalVar'. 
153          also get the new value of the reference variable at the position index indicated by the slider's axis by overwriting the old value. 
154          """
155          if self.listVar:
156              self.globalVar = var
157              var = var[self.xyz]
158          var = float(var)
159          collision = getCollision(self.collision)
160          if not collision:
161              self.pointer.cords[self.xyz] = self.limits[0] + ((var + abs(self.min) * -getSign(self.min)) / self.range) * self.length

163      def drag(self, cIdx: int, dragging: bool):
164          """ 
165          :param cIdx: index of the current hand. 
166          :param dragging: stores if the current controller is pressing the select button. 
167   
168          this method is used to update the position of the slider's pointer if it's being dragged by a hand. 
169   
170          process: 
171              1. if the select button is being held, detect for a collision between the current hand and the slider's pointer. 
172                  set the result of this operation to the 'collision' variable. 
173              2. if there's a collision, set the position of the point to the current hand's position. 
174                  since sliders are 1D, only do this about the axis on which it is summoned. 
175          """
176          self.dragging[cIdx] = dragging
177          if self.dragging[cIdx]:
178              self.collision[cIdx] = detectPointCollision(self.cDat[cIdx].radius, self.pointer.radius, self.cDat[cIdx].cords, self.pointer.cords)
179              if self.collision[cIdx]:
180                  self.pointer.cords[self.xyz] = copy.deepcopy(self.cDat[cIdx].cords[self.xyz])
181          else:
182              self.collision[cIdx] = False

184      def main(self):
185          """ 
186          :return: user-modified value of the reference variable. 
187          """
188          for cIdx in range(controllerCount):
189              resetReferenceVar(self, cIdx)
190              # dismiss the GUI if the close button 'X' is pressed
191              if buttonPressed('select', self.cObj[cIdx], cIdx) and detectPointCollision(self.cDat[cIdx].radius, self.closeButton.radius, self.cDat[cIdx].cords, self.closeButton.cords):
192                  self.unDraw()
193                  self.drawn = False
194  
195          self.boundSlider()  # prevent the pointer from leaving the slider
196  
197          self.pointer.oldCords[self.xyz] += 0.015 * getSign(self.pVelocity[self.xyz]) / renderRate  # adds some drag on the slider
198  
199          collision = getCollision(self.collision)
200          if (abs(self.pVelocity[self.xyz]) < (10 ** -4)) and (not collision):  # if there is to be frictional force on the slider's pointer and velocity is very small, make velocity 0 to prevent vibration
201              self.pointer.oldCords[self.xyz] = copy.deepcopy(self.pointer.cords[self.xyz])
202  
203          # Verlet integration
204          self.pVelocity[self.xyz] = self.pointer.cords[self.xyz] - self.pointer.oldCords[self.xyz]
205          self.pointer.oldCords = copy.deepcopy(self.pointer.cords)
206          self.pointer.cords[self.xyz] += self.pVelocity[self.xyz]
207  
208          self.interpolate()
209  
210          if not self.listVar:
211              return self.var
212          else:
213              tempList = []
214              for axis in range(3):
215                  if axis == self.xyz:
216                      tempList.append(self.var)
217                  else:
218                      tempList.append(self.globalVar[axis])
219              return tempList

221      # prevent the pointer from leaving the slider
222      def boundSlider(self):
223          if self.pointer.cords[self.xyz] < self.limits[0]:
224              self.pointer.cords[self.xyz] = copy.deepcopy(self.limits[0])
225              self.pointer.oldCords[self.xyz] = copy.deepcopy(self.pointer.cords[self.xyz])
226          elif self.pointer.cords[self.xyz] > self.limits[1]:
227              self.pointer.cords[self.xyz] = copy.deepcopy(self.limits[1])
228              self.pointer.oldCords[self.xyz] = copy.deepcopy(self.pointer.cords[self.xyz])

394          this method is used to reset the value of the reference variable by teleporting the slider's pointer and letting the 'interpolate()' method calculate the new value. 
395              if the dial is 3D, change the reference variable with respect to the dial's pointer about all axes. 
396              otherwise, if the dial is 2D, change the reference variable with respect to the dial's pointer about 2 specific axes.

405                  self.p.cords[self.axes[1]] = self.cords[self.axes[1]] + (self.globalOrigVar[self.axes[1]] / self.range[1]) * self.cRad * 2
406          else:
407              if self.tDim:
408                  for axis in range(3):
409                      self.p.cords[axis] = self.cords[axis] + (self.origVar[axis] / self.range[axis]) * self.cRad * 2
410              else:
411                  self.p.cords[self.axes[0]] = self.cords[self.axes[0]] + (self.origVar[self.axes[0]] / self.range[0]) * self.cRad * 2
412                  self.p.cords[self.axes[1]] = self.cords[self.axes[1]] + (self.origVar[self.axes[1]] / self.range[1]) * self.cRad * 2

459      def interpolate(self):
460          relDist = []
461          ratio = []
462          for dim in range(len(self.min)):
463              relDist.append(self.p.cords[self.axes[dim]] - self.cords[self.axes[dim]] + self.cRad)
464              ratio.append(relDist[dim] / (self.cRad * 2))
465              self.var[self.axes[dim]] = self.min[dim] + (self.range[dim] * ratio[dim])

467      def main(self):
468          """ 
469          :return: user-modified value of the reference variable. 
470   
471          this method is used to manage everything for the dial each frame. 
472   
473          process: 
474              1. loop through each hand. update the time since the last double-click of the current hand. 
475                  to save computational power, don't keep updating this value past its limit of 0.25s! 
476              2. if the reset button is pressed by either hand: 
477                  a) if the time since the last reset button press is less than 0.25s, hard reset the reference variable. 
478                  b) otherwise, soft reset the reference variable and reset the time since the last reset button press to 0. 
479              3. if the select button is pressed on the 'X', dismiss the GUI. 
480              4. if the dial's pointer goes outside the circle/sphere, teleport the dial's pointer to its last position from the previous frame. 
481              5. apply drag onto the dial to oppose its pointer's direction of motion. 
482                  if the point is moving super slowly as a result of drag, make its velocity 0 to keep it from constantly oscillating. 
483              6. get the velocity of the point using Verlet integration, and update the position of the point based on its velocity. 
484              7. run the 'interpolate()' method to get the new value of the reference variable based on the pointer's new position in the dial. 
485              8. loop through 'self.axes'. 
486                  this is done to get the selected axes of the GUI to change the value at the position indexes of the vector quantity (list) variable. 
487                  a) if the GUI is 2D, overwrite 'globalVar' at the index position indicated by the dial's axes. 
488                      this is done to allow updating the value of a vector quantity from a 2D dial GUI much easier. 
489                  b) if the GUI is 3D, overwrite all values in 'globalVar'. 
490          """
491          for cIdx in range(controllerCount):
492              resetReferenceVar(self, cIdx)
493              # dismiss the GUI if the close button 'X' is pressed
494              if buttonPressed('select', self.cObj[cIdx], cIdx) and detectPointCollision(self.cDat[cIdx].radius, self.closeButton.radius, self.cDat[cIdx].cords, self.closeButton.cords):
495                  self.unDraw()
496                  self.drawn = False
497  
498          # self.circularMotion()  # MASSIVE WIP so disabled
499          self.boundDial()  # prevent the pointer from leaving the dial
500  
501          # add some drag on the dial's pointer
502          if not self.tDim:
503              movingAngle = getAbsTwoDAngle([0, 0], [self.p.velocity[self.axes[0]], self.p.velocity[self.axes[1]]])
504              self.p.oldCords[self.axes[0]] += 0.015 * getSign(self.p.velocity[self.axes[0]]) * sin(movingAngle) / renderRate
505              self.p.oldCords[self.axes[1]] += 0.015 * getSign(self.p.velocity[self.axes[1]]) * cos(movingAngle) / renderRate
506          else:
507              movingAngle = getAbsThreeDAngle([0, 0, 0], self.p.velocity, 'y')
508              self.p.oldCords[0] += 0.015 * getSign(self.p.velocity[0]) * cos(movingAngle[1]) * sin(movingAngle[0]) / renderRate
509              self.p.oldCords[1] += 0.015 * getSign(self.p.velocity[1]) * sin(movingAngle[1]) / renderRate
510              self.p.oldCords[2] += 0.015 * getSign(self.p.velocity[2]) * cos(movingAngle[1]) * cos(movingAngle[0]) / renderRate
511  
512          collision = getCollision(self.collision)
513          for axis in range(len(self.var)):
514              if (abs(self.p.velocity[axis]) < (10 ** -4)) and (not collision):  # if there is to be frictional force on the dial's pointer and velocity is very small, make velocity 0 to prevent vibrations
515                  self.p.oldCords[axis] = copy.deepcopy(self.p.cords[axis])
516  
517          # Verlet integration
518          for axis in range(3):
519              self.p.velocity[axis] = self.p.cords[axis] - self.p.oldCords[axis]
520          self.p.oldCords = copy.deepcopy(self.p.cords)
521          for axis in range(3):
522              self.p.cords[axis] += self.p.velocity[axis]
523  
524          self.interpolate()
525  
526          tempList = self.globalVar
527          for GUIAxis in self.axes:
528              tempList[GUIAxis] = self.var[GUIAxis]
529          return tempList
530  
531      # MASSIVE WIP METHOD
532      def circularMotion(self):
533          resultAcc = (distance([0, 0, 0], self.p.velocity) ** 2) / self.cRad
534          angle = getAbsTwoDAngle([self.cords[0], self.cords[1]], [self.p.cords[0], self.p.cords[1]])
535          quartiles = [self.p.cords[0] <= self.cords[0], self.p.cords[1] <= self.cords[1]]
536          acc = [resultAcc * sin(angle) * TFNum(quartiles[0]), resultAcc * cos(angle) * TFNum(quartiles[1])]
537          self.p.oldCords[0] -= acc[0]
538          self.p.oldCords[1] -= acc[1]
