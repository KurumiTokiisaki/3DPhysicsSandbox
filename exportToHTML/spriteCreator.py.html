59   def getYCollisionPlane(b, cords):
60       return {
61           'left': {'y': (b.grad['y'] * cords[0]) + (b.vertex[1][1] - (b.grad['y'] * b.vertex[1][0])), 'm': b.grad['y'], 'c': b.vertex[1][1] - (b.grad['y'] * b.vertex[1][0])},
62           'right': {'y': (b.grad['y'] * cords[0]) + (b.vertex[0][1] - (b.grad['y'] * b.vertex[0][0])), 'm': b.grad['y'], 'c': b.vertex[0][1] - (b.grad['y'] * b.vertex[0][0])},
63           'top': {'y': (b.grad['x'] * cords[0]) + (b.vertex[1][1] - (b.grad['x'] * b.vertex[1][0])), 'm': b.grad['x'], 'c': b.vertex[1][1] - (b.grad['x'] * b.vertex[1][0])},
64           'bottom': {'y': (b.grad['x'] * cords[0]) + (b.vertex[7][1] - (b.grad['x'] * b.vertex[7][0])), 'm': b.grad['x'], 'c': b.vertex[7][1] - (b.grad['x'] * b.vertex[7][0])},
65       }

68   def getXCollisionPlane(b, cords):
69       return {
70           'left': {'x': - (b.grad['x'] * cords[1]) + (b.vertex[1][0] + (b.grad['x'] * b.vertex[1][1])), 'm': -b.grad['x'], 'c': b.vertex[1][0] + (b.grad['x'] * b.vertex[1][1])},
71           'right': {'x': - (b.grad['x'] * cords[1]) + (b.vertex[0][0] + (b.grad['x'] * b.vertex[0][1])), 'm': -b.grad['x'], 'c': b.vertex[0][0] + (b.grad['x'] * b.vertex[0][1])},
72           'top': {'x': - (b.grad['y'] * cords[1]) + (b.vertex[1][0] + (b.grad['y'] * b.vertex[1][1])), 'm': -b.grad['y'], 'c': b.vertex[1][0] + (b.grad['y'] * b.vertex[1][1])},
73           'bottom': {'x': - (b.grad['y'] * cords[1]) + (b.vertex[7][0] + (b.grad['y'] * b.vertex[7][1])), 'm': -b.grad['y'], 'c': b.vertex[7][0] + (b.grad['y'] * b.vertex[7][1])}

77   def getCubeCollision(cords, radius, b, yCollisionPlane):
78       return (cords[1] <= (collisionTolerance + yCollisionPlane['top']['y'] + radius / cos(b.angle[2]))) and (cords[1] >= (-collisionTolerance + yCollisionPlane['bottom']['y'] - radius / cos(b.angle[2]))) and (cords[1] <= (collisionTolerance + yCollisionPlane['right']['y'] + radius / sin(b.angle[2]))) and (
79               cords[1] >= (-collisionTolerance + yCollisionPlane['left']['y'] - radius / sin(b.angle[2]))) and (cords[2] <= (collisionTolerance + b.plane['front'] + radius)) and (cords[2] >= (-collisionTolerance + b.plane['back'] - radius))

123      # used to read the contents of exportData to allow the user to import their own creations from spriteCreator.py
124      def __importData(self) -> None:
125          f = open('exportData', 'r')
126          data = f.read().splitlines()
127          formattedData = []  # formattedData = [points, joints, collisionRects]
128          f.close()
129          for pjc in data:
130              if (pjc == 'POINTS') or (pjc == 'JOINTS') or (pjc == 'COLLISIONRECTS'):
131                  formattedData.append([])
132              else:
133                  formattedData[-1].append(pjc)
134                  formattedData[-1][-1] = formattedData[-1][-1].split(' | ')
135          for i in range(len(formattedData)):
136              for j in range(len(formattedData[i])):
137                  for k in range(len(formattedData[i][j])):
138                      if (formattedData[i][j][k] != 's') and (formattedData[i][j][k] != 'l'):
139                          try:
140                              if i == 1:
141                                  formattedData[i][j][k] = int(formattedData[i][j][k])  # should be int for joints since list indexes cannot be floats
142                              else:
143                                  formattedData[i][j][k] = float(formattedData[i][j][k])
144                          except ValueError:
145                              formattedData[i][j][k] = formattedData[i][j][k].replace('[', '')
146                              formattedData[i][j][k] = formattedData[i][j][k].replace(']', '')
147                              tempList = formattedData[i][j][k].split(', ')
148                              for t in range(len(tempList)):
149                                  tempList[t] = float(tempList[t])
150                              formattedData[i][j][k] = tempList
151          points = formattedData[0]
152          joints = formattedData[1]
153          collisionRects = formattedData[2]
154          for p in points:
155              self.addPoint(p[0], p[1])
156              self.points[-1].density = p[2]
157          for j in joints:
158              self.joints.append(Joint(j[0], j[1]))
159          for c in collisionRects:
160              self.addCollisionRect(CollisionRect(c[0], c[1], c[2], c[3], c[4], c[5], c[6], c[7]))

162      # used to import tutorials from 'tutorialTexts.txt'
163      def __importTutorials(self):
164          f = open('tutorialTexts', 'r')
165          tutors = f.read().splitlines()
166          tNames = []
167          tTexts = []
168          for l in tutors:
169              if l.find('---') != -1:
170                  tTexts.append([])
171                  tempList = list(l)
172                  for _ in range(3):
173                      tempList.pop(0)
174                  tempStr = ''
175                  for c in tempList:
176                      tempStr = f'{tempStr}{c}'
177                  tNames.append(tempStr)
178              else:
179                  tTexts[-1].append(l)
180          for t in range(len(tNames)):
181              tNames[t] = tNames[t].replace('newLine', '\n')  # solution from Python Discord, credit goes to lordtyrionlannister "Saul Goodman"
182              self.__tutorialTexts.update({tNames[t]: tTexts[t]})  # update local information about the tutorial
183              tutorialNames.update({tNames[t]: None})  # update the global value of tutorialNames for use in myGUI.GUISelector
184              self.__GUI['Tutorials'][''].update({tNames[t]: None})  # update local value of tutorials in self.GUI
185          f.close()
186          if not imports:
187              self.__GUI['Tutorials']['']['Introduction'] = myGUI.Tutorial([0, 2, 4], [10, 0.2], self.__tutorialTexts['Introduction'], [], 0.3, lControllerObj, rControllerObj)

228      def __dragPoint(self) -> None:
229          # if mode == 'vr':
230          #     print(controlsConf.controllers[0].getButtonState() % touchpad, controlsConf.controllers[1].getButtonState() % touchpad)  # prints the current button being pressed for each controller
231          colliding = False  # True if there is ANY hand<>point/collisionRect collision
232          for c in range(controlsConf.controllerAmt):
233              if self.__clickTime[c] <= 0.25:
234                  self.__clickTime[c] += 1 / calcRate
235              if self.__jClickTime[c] <= 0.25:
236                  self.__jClickTime[c] += 1 / calcRate
237              for g in self.__GUI:
238                  for gu in self.__GUI[g]:
239                      for gui in self.__GUI[g][gu]:
240                          if self.__GUI[g][gu][gui] is not None:
241                              self.__GUI[g][gu][gui].drag(c, selectP(c))  # detect for hand<>GUI interactions
242  
243              # hand<>point dragging
244              for p in range(len(self.points)):
245                  if self.points[p].drawn:
246                      if detectPointCollision(self.points[p].radius, controls.hand[c].radius, self.points[p].cords, controls.hand[c].cords) and (self.dragR[c] is None):
247                          self.points[p].collidingController[c] = True
248                          self.collP[c] = p  # set the collision point to the current point's index
249                          if selectP(c):  # detect if the select button is being pressed, depending on the controller mode
250                              if not self.__buttonHeld['select'][c]:
251                                  self.__buttonHeld['select'][c] = True
252                                  for axis in range(3):
253                                      self.__relPos[c][axis] = self.points[p].cords[axis] - controls.hand[c].cords[axis]
254                                  if self.__clickTime[c] < 0.25:  # if there's a double click, summon sliders (if in VR) or manual inputs (if in keyboard/mouse) to change the density and radius of the double-clicked point
255                                      self.__setRadiusDensityGUI(controls.hand[c].cords, p)
256                                  else:
257                                      self.__clickTime[c] = 0  # reset the time since last click, since this click IS the last click!
258                              if self.dragP[c] is None:  # used to set the drag variables if they are not already set
259                                  self.dragP[c] = p
260                                  if self.lastP[c] != p:  # no need to run the below if the value of lastP won't change
261                                      if mode == 'vr':
262                                          if self.lastP[c - 1] != p:  # prevents both controllers from recalling the same point
263                                              self.lastP[c] = p
264                                      else:
265                                          self.lastP[c] = p
266                      else:
267                          self.points[p].collidingController[c] = False
268  
269                  # make joints when either hand is colliding with a joint
270                  if buttonPressed('dragJoint', controlsConf.controllers[c], c):
271                      collision = detectPointCollision(self.points[p].radius, controls.hand[c].radius, self.points[p].cords, controls.hand[c].cords)
272                      colliding = colliding or collision
273                      if (not self.__buttonHeld['dragJoint'][c]) and collision:  # if either hand is colliding with a point, make a joint
274                          self.__buttonHeld['dragJoint'][c] = True
275                          if self.__jClickTime[c] < 0.25:
276                              self.removeConnectedJoints(p)  # remove all joints connected to the colliding point if 'dragJoint' is double-clicked
277                          else:
278                              self.__jClickTime[c] = 0
279                              if self.__newJoint is None:
280                                  self.__newJoint = Joint(p, c, True)  # attach the joint to the current point and controller's hand
281                              elif self.__newJoint.pOne != p:  # if the point to attach the joint to is not the current point, prevent both points in a joint from being the exact same
282                                  self.__newJoint.pTwo = p
283                                  self.__newJoint.controller = False
284                                  self.joints.append(self.__newJoint)
285                                  self.__newJoint = None
286  
287                  # remove latest created joint
288                  if buttonPressed('undoJoint', controlsConf.controllers[c], c):
289                      if not self.__buttonHeld['undoJoint'][c]:
290                          self.__buttonHeld['undoJoint'][c] = True
291                          self.joints[-1].unDraw()
292                          self.joints.pop(-1)
293  
294              # hand<>collisionRect dragging
295              for cr in range(len(self.collisionRect)):
296                  if self.collisionRect[cr].drawn:
297                      yCollisionPlane = getYCollisionPlane(self.collisionRect[cr], controls.hand[c].cords)
298                      if getCubeCollision(controls.hand[c].cords, controls.hand[c].radius, self.collisionRect[cr], yCollisionPlane) and (self.dragP[c] is None):  # detect for hand<>collisionRect collisions using y = mx + c
299                          self.collisionRect[cr].collidingController[c] = True
300                          if selectP(c):
301                              self.lastR[c] = cr
302                              if not self.__buttonHeld['select'][c]:
303                                  self.__buttonHeld['select'][c] = True
304                                  for axis in range(3):
305                                      self.__relPos[c][axis] = self.collisionRect[cr].cords[axis] - controls.hand[c].cords[axis]
306                                  if self.__clickTime[c] < 0.25:  # if there's a double click, summon sliders (if in VR) or manual inputs (if in keyboard/mouse) to change the size and density of the double-clicked collision rect
307                                      if self.__GUI['GUISelector']['']['pointRect'] is None:
308                                          self.__GUI['GUISelector']['']['pointRect'] = myGUI.GUISelector({'Size': None, 'Solid/\nLiquid': None}, [controls.hand[c].cords[0], controls.hand[c].cords[1] - 1, controls.hand[c].cords[2]], lControllerObj, rControllerObj, cr)
309                                      else:
310                                          self.__GUI['GUISelector']['']['pointRect'].drawn = False
311                                          self.__GUI['GUISelector']['']['pointRect'].unDraw()
312                                  else:
313                                      self.__clickTime[c] = 0  # reset the time since last click, since this click IS the last click!
314                              if self.dragR[c] is None:  # used to set the drag variables if they are not already set
315                                  self.dragR[c] = cr
316                          else:
317                              self.__buttonHeld['select'][c] = False
318                      else:
319                          self.collisionRect[cr].collidingController[c] = False
320              if self.dragP[c] is not None:
321                  for axis in range(3):
322                      self.points[self.dragP[c]].cords[axis] = controls.hand[c].cords[axis] + self.__relPos[c][axis]  # set the point position to the position of the controller that grabbed said point
323              if self.dragR[c] is not None:
324                  tempCords = [0, 0, 0]
325                  for axis in range(3):
326                      tempCords[axis] = controls.hand[c].cords[axis] + self.__relPos[c][axis]
327                  self.collisionRect[self.dragR[c]].cords = copy.deepcopy(tempCords)  # set the collisionRect position to the controller's hand that grabbed said collisionRect's position
328              self.__selectPointAnime(c)
329              # reset drag variables if select button is not pressed
330              if not selectP(c):
331                  self.dragP[c] = None
332                  self.dragR[c] = None
333              # recalls the last clicked point to the controller's position
334              if buttonPressed('recall', controlsConf.controllers[c], c):
335                  if self.lastP[c] is not None:
336                      self.points[self.lastP[c]].cords = copy.deepcopy(controls.hand[c].cords)
337  
338              # remove the joint if 'dragJoint' is pressed again, and the hand isn't colliding with any point
339              # basically just remove the joint if it's being attached to the air
340              if buttonPressed('dragJoint', controlsConf.controllers[c], c) and (self.__newJoint is not None) and (not colliding):
341                  self.__newJoint.unDraw()
342                  self.__newJoint = None

359      def __selectPointAnime(self, c):
360          # unique animation for selecting points
361          if self.collP[c] is not None:  # only run animations if a point is intersecting with a hand
362              if self.dragP[c] is not None:  # only run the below if a point is being dragged
363                  controls.anim[c].point = self.points[self.collP[c]]  # make the selection animation encapsulate the point
364                  if self.__animeScale[c] > (self.points[self.collP[c]].radius / controls.anim[c].sphereRad):  # set the maximum size of the animation equal to the size of the selected point
365                      self.__animeScaleSpeed -= 0.1 / physicsTime
366                      self.__animeScale[c] += self.__animeScaleSpeed
367                      # approximate function for changing color with time based on radius: f(x) = 6 / (time * sqrt(radius * 10))
368                      f = 6 / (renderRate * math.sqrt(self.points[self.collP[c]].radius * 10))
369                      # green-shift the animation
370                      self.__animeColor[c][0] -= f
371                      self.__animeColor[c][1] += f
372                  else:
373                      self.__animeScale[c] = self.points[self.collP[c]].radius / controls.anim[c].sphereRad
374                      controls.anim[c].pause = True
375                  controls.anim[c].setScale(self.__animeScale[c])
376                  controls.anim[c].setColor(self.__animeColor[c])
377              elif not detectPointCollision(self.points[self.collP[c]].radius, controls.hand[c].radius, self.points[self.collP[c]].cords, controls.hand[c].cords):
378                  self.collP[c] = None  # the final else statement takes care of the rest
379              else:
380                  controls.anim[c].point = self.points[self.collP[c]]
381                  self.__animeScale[c] = 1.2 * self.points[self.collP[c]].radius / controls.anim[c].sphereRad
382                  self.__animeScaleSpeed = 0
383                  self.__animeColor[c] = [1, 0, 0]
384                  controls.anim[c].setScale(self.__animeScale[c])
385                  controls.anim[c].setColor(self.__animeColor[c])
386                  controls.anim[c].pause = False
387          else:  # return the animation to the hand. this is a must since points can be deleted.
388              controls.anim[c].point = controls.hand[c]
389              controls.anim[c].resetColor()
390              controls.anim[c].resetScale()
391              self.collP[c] = None
392              controls.anim[c].pause = False

399      def __updateGUI(self):
400          for g in self.__GUI:
401              for gt in self.__GUI[g]:
402                  for gta in self.__GUI[g][gt]:
403                      if self.__GUI[g][gt][gta] is not None:
404                          if self.__GUI[g][gt][gta].drawn:
405                              if g == 'GUISelector':
406                                  self.__GUIType = self.__GUI[g][gt][gta].main()

594      def __export(self):
595          """ 
596          export format: 
597              POINTS: cords, radius, density 
598              JOINTS: pOneIdx, pTwoIdx 
599              COLLISIONRECTS: size, cords, angle, density, viscosity, dragConst, transparency, rectType 
600          """
601          f = open('exportData', 'w')
602          f.write('POINTS\n')
603          for p in self.points:
604              if p.drawn:  # only add points if they haven't been dumped in a 'trash' void box
605                  f.write(f'{p.cords} | {p.radius} | {p.density}\n')
606          f.write('JOINTS\n')
607          for j in self.joints:
608              if j.drawn:  # only add joints if they haven't been dumped in a 'trash' void box
609                  f.write(f'{j.pOne} | {j.pTwo}\n')
610          f.write('COLLISIONRECTS\n')
611          for c in self.collisionRect:
612              if c.drawn:  # only add collisionRects if they haven't been dumped in a 'trash' void box
613                  f.write(f'{[c.size[0] + c.sizeOffset[0], c.size[1] + c.sizeOffset[1], c.size[2] + c.sizeOffset[2]]} | {c.cords} | {c.angle} | {c.density} | {c.viscosity} | {c.dragConst} | {c.transparency} | {c.rectType}\n')
614          f.close()
615          viz.quit()
616          print('Your progress has been saved in exportData!')

646  class CollisionRect:
647      def __init__(self, size, cords, angle, density, viscosity, dragConst, transparency, rectType, *hide):
648          self.drawn = True
649          self.rectType = rectType  # solid or liquid
650          self.angle = angle
651          self.vertexAngle = [0, 0, 0]
652          self.size = size
653          self.origSize = size
654          self.show = False
655          if not hide:
656              self.show = True
657              self.rect = vizshape.addBox([1, 1, 1])
658              self.rect.setScale(self.size)
659          self.cords = copy.deepcopy(cords)
660          self.density = density
661          self.origDensity = density
662          self.dragConst = dragConst
663          self.viscosity = viscosity
664          self.transparency = transparency
665          self.vertex = [None, None, None, None, None, None, None, None]  # [x, y, z] -> [['right', 'top', 'front'], ['left', 'top', 'front'], ['left', 'bottom', 'front'], ['left', 'bottom', 'back'], ['left', 'top', 'back'], ['right', 'top', 'back'], ['right', 'bottom', 'back'], ['right', 'bottom', 'front']]
666          self.plane = {
667              'front': 0,
668              'back': 0,
669              'left': 0,
670              'right': 0,
671              'top': 0,
672              'bottom': 0
673          }
674          self.grad = dict
675          self.sf = 1
676          self.sizeOffset = [0, 0, 0]
677          self.collidingController = [False, False]
678          self.update()
679  
680      def update(self):
681          if self.drawn:
682              tempSize = [self.size[0] + self.sizeOffset[0], self.size[1] + self.sizeOffset[1], self.size[2] + self.sizeOffset[2]]
683              for t in range(3):
684                  if tempSize[t] <= 0:
685                      tempSize[t] = 0.1
686              sizeMultiplier = [0.5, 0.5, 0.5]
687              multiplier = 1
688              self.vertexAngle = math.atan(tempSize[1] / tempSize[0])
689              for v in range(8):
690                  if (v == 1) or (v == 5):
691                      sizeMultiplier[0] = -sizeMultiplier[0]
692                  elif (v == 4) or (v == 6) or (v == 2):
693                      sizeMultiplier[1] = -sizeMultiplier[1]
694                  elif (v == 3) or (v == 7):
695                      sizeMultiplier[2] = -sizeMultiplier[2]
696                  if (v == 1) or (v == 4) or (v == 6) or (v == 7):
697                      multiplier = -1
698                  elif (v == 0) or (v == 5) or (v == 2) or (v == 3):
699                      multiplier = 1
700                  tempVertex = [0, 0, 0]
701                  xySize = math.sqrt((tempSize[0]) ** 2 + (tempSize[1]) ** 2)
702                  for i in range(3):
703                      if i == 0:  # x
704                          tempVertex[i] = self.cords[i] + (xySize * sizeMultiplier[i] * cos(self.vertexAngle + (multiplier * self.angle[2])))
705                      elif i == 1:  # y
706                          tempVertex[i] = self.cords[i] + (xySize * sizeMultiplier[i] * sin(self.vertexAngle + (multiplier * self.angle[2])))
707                      elif i == 2:  # z
708                          tempVertex[i] = self.cords[i] + (tempSize[i] * sizeMultiplier[i])
709  
710                  self.vertex[v] = tempVertex
711  
712              self.plane['right'] = self.cords[0] + (tempSize[0] / 2)
713              self.plane['left'] = self.cords[0] - (tempSize[0] / 2)
714              self.plane['top'] = self.cords[1] + (tempSize[1] / 2)
715              self.plane['bottom'] = self.cords[1] - (tempSize[1] / 2)
716              self.plane['front'] = self.cords[2] + (tempSize[2] / 2)
717              self.plane['back'] = self.cords[2] - (tempSize[2] / 2)
718  
719              if self.vertex[0][0] != self.vertex[1][0]:
720                  mx = (self.vertex[0][1] - self.vertex[1][1]) / (self.vertex[0][0] - self.vertex[1][0])
721              else:
722                  mx = float('inf')
723              if self.vertex[0][0] != self.vertex[7][0]:
724                  my = (self.vertex[0][1] - self.vertex[7][1]) / (self.vertex[0][0] - self.vertex[7][0])
725              else:
726                  my = float('inf')
727              self.grad = {
728                  'x': mx,
729                  'y': my
730              }
731  
732      def draw(self):
733          if self.show and self.drawn:
734              self.update()
735              if self.rectType == 's':
736                  self.transparency = 0.8
737              elif self.rectType == 'l':
738                  self.transparency = 0.5
739              tempSize = [self.size[0] + self.sizeOffset[0], self.size[1] + self.sizeOffset[1], self.size[2] + self.sizeOffset[2]]
740              self.rect.setScale(tempSize)
741              self.rect.setPosition(self.cords)
742              self.rect.setEuler(math.degrees(self.angle[0]), math.degrees(self.angle[1]), math.degrees(self.angle[2]))
743              self.rect.alpha(self.transparency)
744  
745      def unDraw(self):
746          self.rect.remove()
747          self.drawn = False

792          self.collisionRad = 0.25
793          self.text = viz.addText3D('', fontSize=0.1)
794          self.text.setPosition([self.cords[0] - 0.2, self.cords[1] + 0.55, self.cords[2]])
795          if self.spriteType == 'collisionRect':
796              self.innerShape = vizshape.addBox()
797              self.innerShape.setScale([0.5, 0.5, 0.5])
798              self.text.message('Collision Box')
799          elif self.spriteType == 'point':
800              self.innerShape = vizshape.addSphere()
801              self.innerShape.setScale([0.25, 0.25, 0.25])
802              self.text.message('Point')
803          elif self.spriteType == 'trash' and display:
804              self.text.message('Trash Can')
805              self.collisionRad = 0.5
806          if self.spriteType != 'trash':
807              self.innerShape.setPosition(self.cords)
808          self.pHeld = [False, False]

810      def drag(self) -> None:
811          for c in range(controlsConf.controllerAmt):
812              collision = detectPointCollision(controls.hand[c].radius, self.collisionRad, controls.hand[c].cords, self.cords)
813  
814              # if a hand is dragging a point/collisionRect from a void box that summons points/collisionRects
815              if selectP(c):
816                  if not self.pHeld[c]:
817                      self.pHeld[c] = True
818                      if collision:
819                          if self.spriteType == 'point':
820                              game.addPoint(self.cords, 0.1, c)
821                          elif self.spriteType == 'collisionRect':
822                              game.addCollisionRect(CollisionRect([0.25, 0.25, 0.25], self.cords, [0, 0, math.radians(0.001)], 1000, 1, 1, 1, 's'))
823  
824              # if a point/collisionRect is being dropped off by a hand in a trash void box
825              elif self.spriteType == 'trash':
826                  if collision:
827                      if game.lastP[c] is not None:
828                          if game.points[game.lastP[c]].collidingController[c]:
829                              game.removeConnectedJoints(game.lastP[c])  # remove all joints connected to the dismissed point
830                              game.points[game.lastP[c]].unDraw()
831                              game.points.pop(game.lastP[c])
832                              # if the index position of any joint's reference point(s) is >= the index position of the dismissed point, decrease said index position(s) by 1
833                              # this is done since the point object at the index position 'lastP' is popped from the points list, decreasing all subsequent objects' index positions by 1
834                              for j in range(len(game.joints)):
835                                  if game.joints[j].pOne >= game.lastP[c]:
836                                      game.joints[j].pOne -= 1
837                                  if game.joints[j].pTwo >= game.lastP[c]:
838                                      game.joints[j].pTwo -= 1
839                              game.lastP[c] = None
840                              game.dragP[c] = None
841                              game.collP[c] = None
842                      if (game.lastR[c] is not None) and game.collisionRect[game.lastR[c]].collidingController[c]:
843                          game.collisionRect[game.lastR[c]].unDraw()
844              else:
845                  self.pHeld[c] = False
